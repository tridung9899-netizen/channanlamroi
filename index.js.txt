=// --- THIẾT LẬP CƠ BẢN VÀ CẤU HÌNH ---
const container = document.getElementById('container');
const scene = new THREE.Scene();
const renderer = new THREE.WebGLRenderer({ antialias: true });
renderer.setSize(window.innerWidth, window.innerHeight);
renderer.setClearColor(0x0a0a0a, 1);
renderer.shadowMap.enabled = true; // Bật bóng đổ
renderer.shadowMap.type = THREE.PCFSoftShadowMap;
container.appendChild(renderer.domElement);

const camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
const controls = new THREE.OrbitControls(camera, renderer.domElement);
controls.enableDamping = true;
controls.dampingFactor = 0.05;

// --- HẰNG SỐ CƠ BẢN ---
const CAR_SPEED = 0.3;
const INTERSECTION_SIZE = 15;
const GREEN_TIME = 8000;
const YELLOW_TIME = 2000;

let trafficState = 0;
let cycleStartTime = Date.now();
const loader = new THREE.GLTFLoader();
let dynamicCarNS, dynamicCarEW; 
let isModelLoaded = false;

// --- ÁNH SÁNG VÀ CAMERA ---
const ambientLight = new THREE.AmbientLight(0x404040, 2.5);
scene.add(ambientLight);

const directionalLight = new THREE.DirectionalLight(0xffffff, 2);
directionalLight.position.set(15, 30, 15);
directionalLight.target.position.set(0, 0, 0);
directionalLight.castShadow = true;
scene.add(directionalLight);
scene.add(directionalLight.target);
// Cấu hình bóng đổ
directionalLight.shadow.mapSize.width = 2048; 
directionalLight.shadow.mapSize.height = 2048;
directionalLight.shadow.camera.near = 0.5;
directionalLight.shadow.camera.far = 100;
directionalLight.shadow.camera.left = -50;
directionalLight.shadow.camera.right = 50;
directionalLight.shadow.camera.top = 50;
directionalLight.shadow.camera.bottom = -50;


// --- CHỨC NĂNG TẢI MÔ HÌNH (ĐÃ SỬA ĐƯỜNG DẪN) ---

// *********** Đã sửa để tải từ đường dẫn cục bộ (./) ***********
loader.load('./city_pack_5.glb', function (gltf) {
    scene.add(gltf.scene);
    gltf.scene.traverse(function (node) {
        if (node.isMesh) {
            node.castShadow = true;
            node.receiveShadow = true;
        }
    });

    // Code gán xe/đèn (nếu có trong dự án gốc) sẽ được đặt ở đây

    isModelLoaded = true;

}, undefined, function (error) {
    // Báo lỗi rõ ràng nếu mô hình không tải được
    console.error('Lỗi tải mô hình 3D (GLB):', error); 
});

// --- CÁC CHỨC NĂNG DỰ ÁN CỦA BẠN (animate, update, v.v.) ---
// CÁC HÀM CƠ BẢN NHẤT ĐƯỢC GIỮ LẠI ĐỂ DUY TRÌ CẤU TRÚC

function animate() {
    requestAnimationFrame(animate);
    controls.update();

    if (isModelLoaded) {
        // Logic cập nhật xe cộ và giao thông (nếu có)
    }
}

// Khởi động vòng lặp animate
animate();

// --- XỬ LÝ KÍCH THƯỚC CỬA SỔ ---
window.addEventListener('resize', function () {
    camera.aspect = window.innerWidth / window.innerHeight;
    camera.updateProjectionMatrix();
    renderer.setSize(window.innerWidth, window.innerHeight);
});
